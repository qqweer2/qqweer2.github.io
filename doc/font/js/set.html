Set用法


    
        // 创建
        const a = new Set()
        const a = new Set([1,2,3,1])
        Array.from(a); //  [1, 2, 3]
        // 长度
        a.size; // 3
        // 添加
        a.add(4)
        // 删除
        a.delete(1)
        a.clear()
        // 是否包含
        a.has(1)

        // 迭代
        const s = new Set();
        s.add(1).add(2).add(3);
        Array.from(s.keys()); // [1, 2, 3]
        Array.from(s.values()); // [1, 2, 3]
        Array.from(s.entries()); // [[1, 1], [2, 2], [3, 3]]

        for (const i of s) {
            console.log(i);
        } // 1, 2, 3

        s.forEach((value, key) => console.log(key + ' : ' + value));
        // 1 : 1 2 : 2 3 : 3

        // 不支持reduce
