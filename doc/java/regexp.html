分组

用圆括号包裹的部分可以看作一个整体，这个整体作为之表达式，在内部引用\\，在外部引用$， 从1开始

匹配


        // 匹配两个连续的相同数字：(\\d)\\1
        Pattern pattern = Pattern.compile("(\\d)\\1");
        System.out.println(pattern.matcher("11").matches());
        System.out.println(pattern.matcher("12").matches());
        System.out.println(pattern.matcher("121").matches());
        System.out.println(pattern.matcher("122").matches());
        // 匹配个位与千位相同，十位与百位相同的数：(\\d)(\\d)\\2\\1
        Pattern pattern = Pattern.compile("(\\d)(\\d)\\2\\1");
        System.out.println(pattern.matcher("1213").matches());
        System.out.println(pattern.matcher("1221").matches());

格式化


        // 结巴去重
        Pattern pattern = Pattern.compile("(.)\\1+");
        System.out.println(pattern.matcher("我我要学学学编程").replaceAll("$1"));
